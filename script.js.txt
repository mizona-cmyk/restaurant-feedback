const WEB_APP_URL = "https://script.google.com/macros/s/AKfycbzR_TRBUtpP7-rTr3MXms47mrIt8CJ9SH_IMFCPxhJ3daYyGoaCEJ4tY_UQ22N5KnNR/exec";

const form = document.getElementById("feedbackForm");
const message = document.getElementById("message");
const submitBtn = document.getElementById("submitBtn");

form.addEventListener("submit", async function(e){

e.preventDefault();

submitBtn.disabled = true;
submitBtn.innerHTML = "جاري الإرسال...";

const data = {

rating: document.querySelector('input[name="rating"]:checked')?.value || "",

taste: document.getElementById("taste").value,

speed: document.getElementById("speed").value,

clean: document.getElementById("clean").value,

recommend: document.getElementById("recommend").value,

comments: document.getElementById("comments").value,

phone: document.getElementById("phone").value

};

try{

const response = await fetch(WEB_APP_URL,{

method:"POST",

headers:{
"Content-Type":"application/json"
},

body:JSON.stringify(data)

});

const result = await response.json();

if(result.status==="success"){

message.style.color="green";

message.innerHTML="✅ شكراً لك، تم إرسال تقييمك بنجاح.";

form.reset();

}else{

message.style.color="red";

message.innerHTML="❌ حدث خطأ أثناء الإرسال.";

}

}catch(error){

message.style.color="red";

message.innerHTML="❌ تعذر الاتصال بالخادم.";

console.error(error);

}

submitBtn.disabled=false;

submitBtn.innerHTML="إرسال التقييم";

});
